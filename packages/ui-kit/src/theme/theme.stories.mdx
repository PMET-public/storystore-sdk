export const Color = ({ variant = 'surface', ...props }) => (
  <div
    {...props}
    style={{
      padding: 'var(--spacing-sm)',
      background: `var(--color-${variant})`,
      color: `var(--color-on-${variant})`,
      display: 'grid',
      gridGap: 'var(--spacing-sm)',
      ...props.style,
    }}
  >
    <div style={{ fontSize: '0.75em' }}>--color-{variant}</div>
    <div style={{ fontWeight: '600' }}>--color-on-{variant}</div>
  </div>
)

export const Typography = ({ variable, property, ...props }) => {
  const style = { [property]: `var(${variable})` }
  return (
    <div style={{ display: 'grid', gridGap: 'var(--spacing-sm)' }}>
      <div style={{ fontSize: 'var(--font-xm)' }}>{variable}</div>
      <div style={style}>The quick brown fox jumps over the lazy dog</div>
    </div>
  )
}

export const Block = ({ ...props }) => (
  <div
    style={{
      display: 'grid',
      gridGap: '1rem',
      padding: '1rem',
      background: 'var(--color-surface)',
      color: 'var(--color-on-surface)',
    }}
    {...props}
  />
)

export const Spacing = ({ spacing, ...props }) => (
  <div style={{ display: 'grid', gridGap: `var(${spacing})`, gridTemplateColumns: '1fr 1fr' }}>
    <div
      style={{ background: 'var(--color-surface-darker)', height: 50, padding: '0.5rem', fontSize: 'var(--font-xs)' }}
    >
      {spacing}
    </div>
    <div style={{ background: 'var(--color-surface-darker)', height: 50 }} />
  </div>
)
